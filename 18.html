Python Excel读写

周六 03 二月 2018 in Python
#Python

  • xlrd 读模块
  • xlwt 写模块
  • xlutil 修改模块 
import xlwt
import xlrd
# 打开excel，文件必须存在否则报错
book = xlrd.open_workbook('stusinfo.xls')
sheet = book.sheets()
# 获取所有sheet名字，返回list
sheet = book.sheet_names()
print(sheet)
# 获取sheet，通过索引或者名字
sheet = book.sheet_by_index(0)
sheet2 = book.sheet_by_name('stusinfo')
# 获取行内容，返回list
# 读取的数据都是浮点型，需要自己处理下，包括时间类型等等
print(sheet.row_len(0))  # 0行长度
print(sheet.row(0))
print(sheet.row_values(0))
print(sheet2.row_values(1))
print(sheet2.ncols)  # 列数
print(sheet2.nrows)  # 行数
print(book.nsheets)  # sheet数
'''
7
[text:'name', text:'sex', text:'age', text:'addr', text:'grade', text:'phone', text:'gold']
['name', 'sex', 'age', 'addr', 'grade', 'phone', 'gold']
['小黑', '男', 28.0, '河南省济源市北海大道32号', '天蝎座', 14058069895.0, 90988.0]
'''
# 获取列
print(sheet.col(0))
print(sheet.col_values(0))
'''
[text:'name', text:'小黑', text:'小黑', text:'小黑']
['name', '小黑', '小黑', '小黑']
'''
# 获取单元格内容
print(sheet.cell(0, 0))
print(sheet.cell_value(0, 0))
'''
text:'name'
name
'''

# 获取sheet中全部内容
data = []
title = sheet.row_values(0)  # 首行为title
print(title)
for col in range(1, sheet.nrows):
    # 依次遍历其余行，与首行组成dict，拼到data中
    values = sheet.row_values(col)
    data.append(dict(zip(title, values)))
print(data)

# 写入文件
stus = [
    ['姓名', '年龄', '性别', '分数'],
    ['mary', 20, '女', 89.9],
    ['mary', 20, '女', 89.9],
    ['mary', 20, '女', 89.9],
    ['mary', 20, '女', 89.9]
]

# 新建一个excel
book = xlwt.Workbook()
sheet = book.add_sheet('sheet1')  # 添加一个sheet页
# sheet = book.get_sheet('sheet2')  # 获取一个sheet页

row = 0  # 控制行的
for stu in stus:
    col = 0  # 控制列
    for s in stu:
        sheet.write(row, col, s)
        col += 1
    row += 1
book.save('kkk.xls')  # 只有save了才会保存文件

其他

flush_row_data(): 减少内存压力，flush之前行不可再修改

xlutils

copy:       将xlrd.Book转为xlwt.Workbook
styles:     读取xlrd.Workbook的每一个单元格的style
display:    简单而安全地呈现xlrd读取的数据
filter:     拆分与整合多个xls文件
margins:    查看表格稀疏程度
save:       序列化xlrd.Book，转存为binary xls或stream


from xlutils.copy import copy
import xlrd
book1 = xlrd.open_workbook('stu.xls')
book2 = copy(book1)  #拷贝一份原来的excel
sheet = book2.get_sheet(0) #获取第几个sheet页
sheet.write(1,3,0)
sheet.write(1,0,'小黑')
book2.save('stu.xls')
