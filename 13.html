Python time模块

周四 18 一月 2018 in Python
#Python 

"""
# 1、格式化好的时间 2018-1-14 16:42
# 2、时间戳 是从unix元年到现在所有的秒数
# 3、时间元组
"""
import time

#当前时间戳
print(time.time())#浮点型
print(int(time.time()))

#标准时区的时间元祖，接受时间戳参数，可以返回时间元祖
print(time.gmtime())
#本地时区当前时间元祖，接受时间戳参数，可以返回时间元祖
t1 = time.time()
print(time.localtime())
print(time.localtime(t1))

#当前时间格式化输出，可以接受时间元祖参数，返回格式化的时间，必须要有格式化参数
print(time.strftime('%Y-%m-%d %H:%M:%S'))
t1 = time.localtime()
print(time.strftime('%Y-%m-%d %H:%M:%S',t1))

#将格式化的时间转成时间元祖
print(time.strftime('%Y-%m-%d %H:%M:%S'))
t2 = time.strftime('%Y-%m-%d %H:%M:%S')
print(time.strptime(t2,'%Y-%m-%d %H:%M:%S'))

#将时间元祖转成时间戳，浮点型
t1 = time.localtime()
print(t1)
print(time.mktime(t1))

#已经格式化好的当前时间
print(time.asctime(t1))

def timestampToStr(time_strmp,format='%Y%m%d%H%M%S'):
    """时间戳转格式化好的时间"""
    cur_time = time.localtime(time_strmp)
    res = time.strftime(format, cur_time)
    return res

def strToTimestamp(time_st,format='%Y%m%d%H%M%S'):
    """这个函数是格式化好的时间，转时间戳的"""
    t=time.strptime(time_st,format)
    res = time.mktime(t)
    return res
