#Python

集合是一个无序不重复元素的集。基本功能包括关系测试和消除重复元素。集合对象还支持 union（联合），intersection（交），difference（差）和 sysmmetric difference（对称差集）等数学运算。

myset = set([1, 2, 3, 'A', 'B', 'C', 1, 2, 3])
myset2 = set([1, 2, 'B', 'C', 'E'])

# 查看,去重
print(myset)
{1, 2, 3, 'A', 'B', 'C'}
# 添加
myset.add('D')
print(myset)
{1, 2, 3, 'A', 'D', 'B', 'C'}
# 删除
myset.remove('D')
print(myset)
{1, 2, 3, 'A', 'B', 'C'}

# 移除最早值
myset.pop()
print(myset)
{2, 3, 'A', 'B', 'C'}

# 并集
#myset | myset2
print('并集',myset | myset2)
并集 {1, 2, 3, 'A', 'E', 'B', 'C'}

# 交集
#myset & myset2
print('交集',myset & myset2)
并集 {1, 2, 3, 'A', 'E', 'B', 'C'}

# 差集
#myset - myset2
print('差集',myset - myset2)
差集 {3, 'A'}
print('差集',myset2 - myset)
差集 {1, 'E'}

# 对称差集（项在myset或myset2中，但不会同时出现在二者中）
print('对称差集',myset ^ myset2)
对称差集 {1, 3, 'A', 'E'}


# 遍历

for v in myset:
    print(v)

for i, v in enumerate(myset):
    print("index = %s，value = %s" % (i, v))
index = 0，value = 2
index = 1，value = 3
index = 2，value = A
index = 3，value = B
index = 4，value = C
